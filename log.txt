GetDelta(axis){
    if (axis < 40){
        delta := axis - 40
    }else if (axis > 60){
        delta := axis - 60
    }else{
        delta := 0
    }
    return delta
}

WatchJoystick:
    X := GetKeyState("JoyU")
    Y := GetKeyState("JoyR")

    deltaX := Getdelta(X)
    deltaY := Getdelta(Y)

    if abs(deltaX) > 0 or abs(deltaY) > 0
                DllCall("mouse_event" ,uint, 1, int, deltaX*0.3, int, deltaY*0.3)
